            // ê¸ˆì£¼: ì´ë²ˆ ì£¼ ì›”ìš”ì¼ ~ ì¼ìš”ì¼

            end = getSunday(today);

function groupByEmployee(reports) {

        grouped[employee].subtotal.targetSales += Number(report.targetSalesAmount) || 0;
