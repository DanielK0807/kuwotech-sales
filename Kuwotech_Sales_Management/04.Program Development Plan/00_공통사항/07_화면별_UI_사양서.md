## 🎯 7. 실제 구현 예시

### 7.1 동적 레이아웃 매니저

```javascript
class LayoutManager {
  constructor() {
    this.layouts = new Map();
    this.currentLayout = null;
    this.breakpoint = this.getCurrentBreakpoint();
    
    this.init();
  }
  
  init() {
    // 레이아웃 등록
    this.registerLayout('login', LoginPageLayout);
    this.registerLayout('admin', AdminDashboardLayout);
    this.registerLayout('sales', SalesDashboardLayout);
    
    // 브레이크포인트 리스너
    this.listenToResize();
    
    // 초기 레이아웃 적용
    this.applyLayout();
  }
  
  registerLayout(name, config) {
    this.layouts.set(name, config);
  }
  
  getCurrentBreakpoint() {
    const width = window.innerWidth;
    
    for (const [name, config] of Object.entries(ResponsiveBreakpoints)) {
      if (
        (!config.minWidth || width >= config.minWidth) &&
        (!config.maxWidth || width <= config.maxWidth)
      ) {
        return name;
      }
    }
    
    return 'lg'; // 기본값
  }
  
  applyLayout() {
    const breakpointConfig = ResponsiveBreakpoints[this.breakpoint];
    const layout = breakpointConfig.layout;
    
    // CSS 변수 업데이트
    Object.entries(layout).forEach(([key, value]) => {
      if (typeof value === 'string' || typeof value === 'number') {
        const cssVar = `--${this.kebabCase(key)}`;
        document.documentElement.style.setProperty(cssVar, value);
      }
    });
    
    // 레이아웃별 클래스 적용
    document.body.className = `breakpoint-${this.breakpoint}`;
    
    // 커스텀 이벤트 발생
    this.emitLayoutChange();
  }
  
  listenToResize() {
    let resizeTimer;
    
    window.addEventListener('resize', () => {
      clearTimeout(resizeTimer);
      resizeTimer = setTimeout(() => {
        const newBreakpoint = this.getCurrentBreakpoint();
        
        if (newBreakpoint !== this.breakpoint) {
          this.breakpoint = newBreakpoint;
          this.applyLayout();
        }
      }, 150);
    });
  }
  
  emitLayoutChange() {
    window.dispatchEvent(new CustomEvent('layoutchange', {
      detail: {
        breakpoint: this.breakpoint,
        layout: ResponsiveBreakpoints[this.breakpoint].layout
      }
    }));
  }
  
  kebabCase(str) {
    return str.replace(/([a-z])([A-Z])/g, '$1-$2').toLowerCase();
  }
}

// 사용 예시
const layoutManager = new LayoutManager();

// 레이아웃 변경 감지
window.addEventListener('layoutchange', (e) => {
  console.log('Layout changed:', e.detail);
  
  // 컴포넌트 재렌더링
  if (e.detail.breakpoint === 'xs' || e.detail.breakpoint === 'sm') {
    // 모바일 레이아웃 적용
    enableMobileMenu();
    convertToCardView();
  } else {
    // 데스크톱 레이아웃 적용
    disableMobileMenu();
    convertToTableView();
  }
});
```
