## 3. 페이지 전환 효과

### 3.1 역할별 전환 효과

```javascript
class PageTransition {
  constructor() {
    this.currentTheme = document.body.className.match(/theme-(\w+)/)?.[1] || 'login';
  }
  
  // 역할별 전환 설정
  getTransitionConfig() {
    const configs = {
      login: {
        enter: 'fadeSlideUp',
        exit: 'fadeSlideDown',
        duration: 400
      },
      sales: {
        enter: 'slideRotate',
        exit: 'slideScale',
        duration: 350
      },
      admin: {
        enter: 'smoothFade',
        exit: 'smoothScale',
        duration: 450
      }
    };
    
    return configs[this.currentTheme];
  }
  
  // 페이지 진입 애니메이션
  async enter(element) {
    const config = this.getTransitionConfig();
    element.classList.add(`page-${config.enter}`);
    
    await this.wait(config.duration);
    element.classList.remove(`page-${config.enter}`);
  }
  
  // 페이지 나가기 애니메이션
  async exit(element) {
    const config = this.getTransitionConfig();
    element.classList.add(`page-${config.exit}`);
    
    await this.wait(config.duration);
    element.classList.remove(`page-${config.exit}`);
  }
  
  wait(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

## 5. 스크롤 인터랙션

### 5.1 스크롤 트리거 애니메이션

```javascript
class ScrollAnimations {
  constructor() {
    this.observer = null;
    this.animatedElements = new Set();
    this.init();
  }
  
  init() {
    this.setupObserver();
    this.observeElements();
  }
  
  setupObserver() {
    const options = {
      root: null,
      rootMargin: '0px',
      threshold: [0, 0.1, 0.25, 0.5, 0.75, 1]
    };
    
    this.observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        const element = entry.target;
        const ratio = entry.intersectionRatio;
        
        if (entry.isIntersecting) {
          // 스크롤 기반 애니메이션
          this.animateOnScroll(element, ratio);
          
          // 한 번만 실행되는 애니메이션
          if (element.dataset.animateOnce && !this.animatedElements.has(element)) {
            this.animateOnce(element);
            this.animatedElements.add(element);
          }
        } else {
          // 요소가 뷰포트를 벗어났을 때
          if (!element.dataset.animateOnce) {
            this.resetAnimation(element);
          }
        }
      });
    }, options);
  }
  
  observeElements() {
    // 스크롤 애니메이션 요소들 관찰
    document.querySelectorAll('[data-scroll-animate]').forEach(el => {
      this.observer.observe(el);
    });
  }
  
  animateOnScroll(element, ratio) {
    const animation = element.dataset.scrollAnimate;
    
    switch(animation) {
      case 'fadeIn':
        element.style.opacity = ratio;
        break;
        
      case 'slideUp':
        element.style.transform = `translateY(${(1 - ratio) * 30}px)`;
        element.style.opacity = ratio;
        break;
        
      case 'scaleIn':
        element.style.transform = `scale(${0.8 + (ratio * 0.2)})`;
        element.style.opacity = ratio;
        break;
        
      case 'glassIn':
        element.style.backdropFilter = `blur(${ratio * 12}px)`;
        element.style.opacity = ratio;
        break;
        
      case 'parallax':
        const speed = element.dataset.parallaxSpeed || 0.5;
        element.style.transform = `translateY(${(1 - ratio) * 100 * speed}px)`;
        break;
    }
  }
  
  animateOnce(element) {
    element.classList.add('animated');
    
    // 스태거 애니메이션
    if (element.dataset.stagger) {
      const children = element.children;
      Array.from(children).forEach((child, index) => {
        child.style.animationDelay = `${index * 100}ms`;
        child.classList.add('stagger-item');
      });
    }
  }
  
  resetAnimation(element) {
    element.style.opacity = '';
    element.style.transform = '';
    element.style.backdropFilter = '';
  }
}
```

### 9.1 애니메이션 최적화

```javascript
class AnimationOptimizer {
  constructor() {
    this.prefersReducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;
    this.isLowPerformance = this.detectLowPerformance();
  }
  
  detectLowPerformance() {
    // 디바이스 성능 감지
    const memory = navigator.deviceMemory || 4;
    const cores = navigator.hardwareConcurrency || 2;
    
    return memory < 4 || cores < 4;
  }
  
  optimizeAnimations() {
    if (this.prefersReducedMotion) {
      // 모션 감소 설정
      document.documentElement.style.setProperty('--duration-fast', '0ms');
      document.documentElement.style.setProperty('--duration-normal', '0ms');
      document.documentElement.style.setProperty('--duration-slow', '0ms');
    }
    
    if (this.isLowPerformance) {
      // 저성능 디바이스 최적화
      document.documentElement.style.setProperty('--glass-blur', 'blur(4px)');
      document.documentElement.style.setProperty('--glass-blur-heavy', 'blur(8px)');
      
      // 복잡한 애니메이션 비활성화
      document.body.classList.add('reduce-animations');
    }
  }
  
  // GPU 가속 강제
  forceGPUAcceleration(element) {
    element.style.transform = 'translateZ(0)';
    element.style.willChange = 'transform, opacity';
  }
  
  // 애니메이션 완료 후 정리
  cleanupAnimation(element) {
    element.style.willChange = 'auto';
  }
}
```

### 9.2 성능 모니터링

```javascript
class PerformanceMonitor {
  constructor() {
    this.frameRate = 60;
    this.lastTime = performance.now();
    this.frames = 0;
  }
  
  measureFrameRate() {
    const now = performance.now();
    this.frames++;
    
    if (now >= this.lastTime + 1000) {
      this.frameRate = Math.round((this.frames * 1000) / (now - this.lastTime));
      this.frames = 0;
      this.lastTime = now;
      
      // 프레임레이트가 낮으면 애니메이션 간소화
      if (this.frameRate < 30) {
        this.reduceAnimationQuality();
      }
    }
    
    requestAnimationFrame(() => this.measureFrameRate());
  }
  
  reduceAnimationQuality() {
    console.warn(`낮은 프레임레이트 감지: ${this.frameRate}fps`);
    document.body.classList.add('low-fps');
  }
}
```
